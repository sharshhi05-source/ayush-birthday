<script>
  var app = angular.module('wishApp', []);

  app.controller('wishCtrl', function($scope, $sce, $interval) {
      $scope.openPage = function(page) {
          $scope.currentPage = $sce.trustAsResourceUrl(page);
      }

      $scope.balloons = [];
      function releaseBalloon() {
        let x = Math.random() * window.innerWidth;
        $scope.balloons.push({style:{left: x+'px'}});
      }
      $interval(releaseBalloon, 800);

      $scope.blast = function(event) {
        let blast = document.createElement('div');
        blast.className = 'blast';
        blast.style.left = event.clientX + 'px';
        blast.style.top = event.clientY + 'px';
        document.body.appendChild(blast);
        setTimeout(()=>blast.remove(),1000);
      }
  });
</script>
